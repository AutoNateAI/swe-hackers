import openai
from functools import lru_cache

class OracleEngine:
    def __init__(self, api_key: str):
        self.client = openai.OpenAI(api_key=api_key)
        self.history = []
    
    def build_prompt(self, template: str, **kwargs) -> str:
        """Fill in prompt template with variables."""
        return template.format(**kwargs)
    
    @lru_cache(maxsize=100)  # Cache responses!
    def query(self, prompt: str, system_message: str = "") -> str:
        """Send query to AI and get response."""
        messages = []
        if system_message:
            messages.append({"role": "system", "content": system_message})
        messages.append({"role": "user", "content": prompt})
        
        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=messages,
            max_tokens=1000
        )
        
        result = response.choices[0].message.content
        self._store_history(prompt, result)
        return result
    
    def _store_history(self, prompt, response):
        """Store interaction for later review."""
        self.history.append({
            "prompt": prompt,
            "response": response,
            "timestamp": datetime.now().isoformat()
        })
